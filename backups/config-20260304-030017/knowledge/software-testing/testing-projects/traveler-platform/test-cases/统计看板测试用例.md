### 2. 自动化分层

**Layer 1: 页面对象模型（POM）**
```javascript
// pages/DashboardPage.js
class DashboardPage {
  constructor(page) {
    this.page = page;
    this.menuDashboard = 'text=统计看板';
    this.refreshButton = 'button:has-text("刷新")';
    this.timeSelector = '[data-test="time-selector"]';
    this.exportButton = 'button:has-text("导出")';
  }

  async navigate() {
    await this.page.click(this.menuDashboard);
    await this.page.waitForLoadState('networkidle');
  }

  async refresh() {
    await this.page.click(this.refreshButton);
    await this.page.waitForTimeout(2000);
  }

  async selectTimeRange(range) {
    await this.page.selectOption(this.timeSelector, range);
    await this.page.waitForTimeout(1000);
  }
}
```
