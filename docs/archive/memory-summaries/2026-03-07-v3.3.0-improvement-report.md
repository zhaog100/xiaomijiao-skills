  ```bash
  DEBOUNCE_SECONDS=20

  check_debounce() {
      local now=$(date +%s)
      local elapsed=$((now - LAST_RUN))
      if [ $elapsed -lt $DEBOUNCE_SECONDS ]; then
          log "⏸️ 防抖中（剩余 ${remaining}秒），跳过"
          return 1
      fi
  }
  ```
  
- ✅ **PID 锁机制**
  ```bash
  try_acquire_lock() {
      # 检查进程是否存活
      if kill -0 "$pid" 2>/dev/null; then
          log "🔒 另一个进程正在运行，跳过"
          return 1
      else
          log "🧹 清理僵尸锁"
          rm -f "$LOCK_FILE"
      fi
      # 获取锁
      echo $$ > "$LOCK_FILE"
  }
  ```

**效果**：

#### 2. 去重检查（.processed 标记）⭐⭐⭐⭐⭐

**文件**：`session-memory-enhanced-v3.3.sh`（6.8KB）

**改进内容**：
- ✅ **固化时标记**
  ```bash
  flush_tail() {
      # 生成 .processed 标记文件
      local processed_marker="$part_file.processed"
      
      # 去重检查
      if [ -f "$processed_marker" ]; then
          log "✅ 分片已处理，跳过"
          return 0
      fi
      
      # 处理后标记
      touch "$processed_marker"
  }
  ```

- ✅ **QMD 去重**
  ```bash
  update_qmd() {
      for part_file in "$MEMORY_DIR"/part*.json; do
          if [ -f "$part_file.processed" ]; then
              skipped_count++  # 跳过已处理
              continue
          fi
          # 处理新分片
          touch "$part_file.processed"
      done
  }
  ```
