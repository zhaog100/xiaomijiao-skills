## 🛠️ 测试工具栈

### 1. Appium（主流工具）⭐⭐⭐⭐⭐

**简介**：
- 跨平台移动自动化测试框架
- 支持 iOS 和 Android
- 支持原生应用、混合应用、移动Web应用
- 使用 WebDriver 协议

**核心优势**：
- ✅ 跨平台（一套代码，多平台运行）
- ✅ 多语言支持（Python、Java、JavaScript、Ruby等）
- ✅ 不依赖应用源码
- ✅ 社区活跃，文档完善

**安装**：
```bash
# 安装 Node.js（必需）
npm install -g appium

# 安装 Appium Doctor（环境检查）
npm install -g appium-doctor

# 检查环境
appium-doctor

# 安装客户端库（Python）
pip install Appium-Python-Client

# 安装客户端库（JavaScript）
npm install webdriverio
```

**基础示例**（Python）：
```python
from appium import webdriver

# Android 配置
desired_caps = {
    'platformName': 'Android',
    'platformVersion': '12',
    'deviceName': 'Android Emulator',
    'app': '/path/to/app.apk',
    'automationName': 'UiAutomator2'
}

# iOS 配置
desired_caps = {
    'platformName': 'iOS',
    'platformVersion': '16.0',
    'deviceName': 'iPhone 14',
    'app': '/path/to/app.app',
    'automationName': 'XCUITest'
}

# 连接设备
driver = webdriver.Remote('http://localhost:4723/wd/hub', desired_caps)

# 基本操作
element = driver.find_element('id', 'button_id')
element.click()

# 输入文本
text_field = driver.find_element('id', 'text_field_id')
text_field.send_keys('Hello World')

# 滑动操作
driver.swipe(100, 500, 100, 100, 800)

# 截图
driver.save_screenshot('screenshot.png')

# 关闭
driver.quit()
```

**核心优势**：
- ✅ 性能优秀

**基础示例**（Java）：
```java
// 添加依赖（build.gradle）
dependencies {
    androidTestImplementation 'androidx.test.espresso:espresso-core:3.5.1'
}

// 测试代码
@RunWith(AndroidJUnit4.class)
public class ExampleTest {
    @Rule
    public ActivityScenarioRule<MainActivity> activityRule =
        new ActivityScenarioRule<>(MainActivity.class);

    @Test
    public void testButtonClick() {
        // 点击按钮
        onView(withId(R.id.button))
            .perform(click());

        // 验证文本
        onView(withId(R.id.text_view))
            .check(matches(withText("Button Clicked")));
    }
}
```

### 3. XCUITest（iOS原生）

**简介**：
- Apple 官方 iOS UI 测试框架
- 只支持 iOS
- 与 Xcode 深度集成

**核心优势**：
- ✅ 性能优秀
- ✅ 官方支持
- ✅ 稳定性高

**基础示例**（Swift）：
```swift
import XCTest

class ExampleTest: XCTestCase {
    var app: XCUIApplication!

    override func setUp() {
        super.setUp()
        app = XCUIApplication()
        app.launch()
    }

    func testButtonClick() {
        // 点击按钮
        let button = app.buttons["button_id"]
        button.tap()

        // 验证文本
        let textView = app.staticTexts["text_view_id"]
        XCTAssertEqual(textView.label, "Button Clicked")
    }
}
```

**基础示例**（Dart）：
```dart
// 测试代码
void main() {
  group('Example Test', () {
    FlutterDriver driver;

    setUpAll(() async {
      driver = await FlutterDriver.connect();
    });

    tearDownAll(() async {
      if (driver != null) {
        await driver.close();
      }
    });

    test('tap button', () async {
      await driver.tap(find.byValueKey('button'));
      await driver.waitFor(find.text('Button Clicked'));
    });

}
```

## 📝 测试框架设计

### POM 设计模式（移动端）

```python
# base_page.py
class BasePage:
    def __init__(self, driver):
        self.driver = driver

    def find_element(self, locator):
        return self.driver.find_element(*locator)

    def click(self, locator):
        self.find_element(locator).click()

    def send_keys(self, locator, text):
        self.find_element(locator).send_keys(text)

    def is_element_present(self, locator):
        try:
            self.find_element(locator)
            return True
        except:
            return False

# login_page.py
from base_page import BasePage

class LoginPage(BasePage):
    # 定位器
    USERNAME_FIELD = ('id', 'username')
    PASSWORD_FIELD = ('id', 'password')
    LOGIN_BUTTON = ('id', 'login_button')
    ERROR_MESSAGE = ('id', 'error_message')

    # 操作方法
    def login(self, username, password):
        self.send_keys(self.USERNAME_FIELD, username)
        self.send_keys(self.PASSWORD_FIELD, password)
        self.click(self.LOGIN_BUTTON)

    def get_error_message(self):
        return self.find_element(self.ERROR_MESSAGE).text

# test_login.py
import pytest
from login_page import LoginPage

class TestLogin:
    @pytest.fixture(autouse=True)
    def setup(self, driver):
        self.login_page = LoginPage(driver)

    def test_login_success(self):
        self.login_page.login('test_user', 'test_password')
        assert self.login_page.is_element_present(('id', 'welcome_text'))

    def test_login_failure(self):
        self.login_page.login('wrong_user', 'wrong_password')
        error_message = self.login_page.get_error_message()
        assert 'Invalid credentials' in error_message
```

## 🎯 最佳实践

### 1. 元素定位策略

**优先级排序**：
1. **Accessibility ID**（推荐）⭐⭐⭐⭐⭐
   - iOS：`accessibilityIdentifier`
   - Android：`content-desc`

2. **ID**（推荐）⭐⭐⭐⭐
   - iOS：`resource-id`
   - Android：`resource-id`

3. **Class Name**（一般）⭐⭐⭐
   - 不够唯一，容易重复

4. **XPath**（不推荐）⭐⭐
   - 速度慢，易碎

**示例**：
```python
# 推荐：Accessibility ID
element = driver.find_element('accessibility id', 'login_button')

# 推荐：ID
element = driver.find_element('id', 'com.example.app:id/login_button')

# 不推荐：XPath
element = driver.find_element('xpath', '//android.widget.Button[@text="Login"]')
```

### 2. 等待策略

**显式等待**（推荐）：
```python
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# 等待元素出现
element = WebDriverWait(driver, 10).until(
    EC.presence_of_element_located(('id', 'button'))
)

# 等待元素可点击
element = WebDriverWait(driver, 10).until(
    EC.element_to_be_clickable(('id', 'button'))
)

# 等待元素消失
WebDriverWait(driver, 10).until_not(
    EC.presence_of_element_located(('id', 'loading'))
)
```

**隐式等待**（不推荐）：
```python
driver.implicitly_wait(10)
```

### 4. 失败截图

```python
import pytest
from datetime import datetime

@pytest.hookimpl(hookwrapper=True)
def pytest_runtest_makereport(item, call):
    outcome = yield
    report = outcome.get_result()

    if report.when == 'call' and report.failed:
        # 截图
        driver = item.funcargs.get('driver')
        if driver:
            timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
            screenshot_path = f'screenshots/{item.name}_{timestamp}.png'
            driver.save_screenshot(screenshot_path)
```
