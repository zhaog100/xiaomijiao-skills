### 1. PyAutoGUI（跨平台）⭐⭐⭐⭐⭐

**简介**：
- Python 跨平台 GUI 自动化工具
- 支持 Windows、macOS、Linux
- 基于坐标和图像识别

**核心优势**：
- ✅ 跨平台
- ✅ 简单易用
- ✅ 基于图像识别，不依赖应用内部

**安装**：
```bash
pip install pyautogui
pip install pillow  # 图像处理
pip install opencv-python  # 图像识别增强
```

**基础示例**：
```python
import pyautogui
import time

# 安全设置（鼠标移到屏幕角落会停止）
pyautogui.FAILSAFE = True

# 获取屏幕尺寸
screen_width, screen_height = pyautogui.size()
print(f'Screen: {screen_width}x{screen_height}')

# 鼠标操作
# 移动鼠标（绝对坐标）
pyautogui.moveTo(100, 100, duration=1)

# 移动鼠标（相对坐标）
pyautogui.move(50, 50, duration=1)

# 点击
pyautogui.click(100, 100)  # 左键点击
pyautogui.rightClick(100, 100)  # 右键点击
pyautogui.doubleClick(100, 100)  # 双击

# 拖拽
pyautogui.drag(100, 100, duration=1, button='left')

# 滚动
pyautogui.scroll(200)  # 向上滚动
pyautogui.scroll(-200)  # 向下滚动

# 键盘操作
# 输入文本
pyautogui.typewrite('Hello World', interval=0.1)

# 按键
pyautogui.press('enter')
pyautogui.press('tab')

# 组合键
pyautogui.hotkey('ctrl', 'c')  # Ctrl+C
pyautogui.hotkey('ctrl', 'v')  # Ctrl+V
pyautogui.hotkey('alt', 'f4')  # Alt+F4

# 截图
screenshot = pyautogui.screenshot()
screenshot.save('screenshot.png')

# 图像识别
# 查找图像位置
button_location = pyautogui.locateOnScreen('button.png')
if button_location:
    center = pyautogui.center(button_location)
    pyautogui.click(center)

# 等待图像出现（最多等待10秒）
button_location = pyautogui.locateOnScreen('button.png', confidence=0.9, timeout=10)

# 模糊匹配（confidence 0-1）
button_location = pyautogui.locateOnScreen('button.png', confidence=0.8)
```

```python
import psutil
import time

def test_performance():
    # 测试启动时间
    start_time = time.time()

    # 启动应用
    app = Application().start('myapp.exe')
    window = app.window(title='MyApp')

    end_time = time.time()
    launch_time = end_time - start_time
    print(f'Launch time: {launch_time:.2f}s')
    assert launch_time < 5, f'Launch time {launch_time:.2f}s exceeds 5s'

    # 测试内存使用
    process = psutil.Process(app.process)
    memory_info = process.memory_info()
    print(f'Memory usage: {memory_info.rss / 1024 / 1024:.2f} MB')

    print(f'CPU usage: {cpu_percent:.2f}%')

    # 测试响应时间
    start_time = time.time()
    button = window['Button1']
    button.click()
    end_time = time.time()
    response_time = end_time - start_time
    print(f'Response time: {response_time:.2f}s')
    assert response_time < 2, f'Response time {response_time:.2f}s exceeds 2s'

    # 关闭应用
    window.close()

test_performance()
```

### POM 设计模式（桌面端）

```python
# base_page.py
class BasePage:
    def __init__(self, app):
        self.app = app
        self.window = None

    def find_element(self, locator):
        if isinstance(locator, tuple):
            locator_type, locator_value = locator
            if locator_type == 'title':
                return self.window.window(title=locator_value)
            elif locator_type == 'auto_id':
                return self.window.child_window(auto_id=locator_value)
            elif locator_type == 'class':
                return self.window.child_window(class_name=locator_value)
        else:
            return self.window[locator]

    def click(self, locator):
        self.find_element(locator).click()

    def type_keys(self, locator, text):
        self.find_element(locator).type_keys(text, with_spaces=True)

    def get_text(self, locator):
        return self.find_element(locator).window_text()

    def is_element_present(self, locator):
        try:
            self.find_element(locator)
            return True
        except:
            return False

# login_page.py
from base_page import BasePage

class LoginPage(BasePage):
    # 定位器
    USERNAME_FIELD = ('auto_id', 'username_textbox')
    PASSWORD_FIELD = ('auto_id', 'password_textbox')
    LOGIN_BUTTON = ('auto_id', 'login_button')
    ERROR_MESSAGE = ('auto_id', 'error_label')

    # 操作方法
    def login(self, username, password):
        self.type_keys(self.USERNAME_FIELD, username)
        self.type_keys(self.PASSWORD_FIELD, password)
        self.click(self.LOGIN_BUTTON)

    def get_error_message(self):
        return self.get_text(self.ERROR_MESSAGE)

# test_login.py
import pytest
```

## 🎯 最佳实践

### 1. 元素定位策略

**优先级排序**：
1. **Automation ID**（推荐）⭐⭐⭐⭐⭐
   - 唯一且稳定

2. **标题/文本**（推荐）⭐⭐⭐⭐
   - 容易理解，但可能变化

3. **类名**（一般）⭐⭐⭐
   - 不够唯一

4. **坐标**（不推荐）⭐⭐
   - 分辨率变化会失效

**示例**：
```python
# 推荐：Automation ID
element = window.child_window(auto_id='login_button')

# 推荐：标题
element = window.window(title='Login')

# 不推荐：坐标
pyautogui.click(100, 100)
```

```python
Timings.fast()

element.wait('ready', timeout=5)

# 自定义等待
def wait_for_element(window, locator, timeout=10):
    start_time = time.time()
    while time.time() - start_time < timeout:
        try:
            element = window.child_window(**locator)
            if element.exists():
                return element
        except:
            pass
        time.sleep(0.5)
    raise TimeoutError(f'Element {locator} not found after {timeout}s')
```

```python
@pytest.hookimpl(hookwrapper=True)
```
